<script>
    function updateTime(){
        var now=new Date();
        var weekDays=['日','一','二','三','四','五','六'];
        document.getElementById('datetime').textContent=now.getFullYear()+'年'+(now.getMonth()+1)+'月'+now.getDate()+'日 '+now.getHours().toString().padStart(2,'0')+':'+now.getMinutes().toString().padStart(2,'0');
        document.getElementById('weekday').textContent='星期'+weekDays[now.getDay()];
    }
    updateTime();
    setInterval(updateTime,60000); // 每分钟更新一次即可
</script>
